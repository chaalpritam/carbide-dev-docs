Quick Start

Swift 
import CarbideSDK

// Initialize the client
let client = CarbideClient()

// Upload a file with encryption
let result = try await client.uploadFile(
    from: fileURL,
    preferredRegion: .northAmerica,
    encrypt: true
)

print("File ID: \(result.fileID)")
